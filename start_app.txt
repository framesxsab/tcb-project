# Instructions to start the app

# 1. Install dependencies
pip install -r requirements.txt

# 2. Preprocess the data (if needed)
cd src
python preprocess.py

# 3. Run the Streamlit app
# Option 1: Run app.py (uses online EMID dataset)
streamlit run app.py

# Option 2: Run app1.py (uses local spotify_millsongdata.csv dataset)
streamlit run app1.py

# Alternative: If using virtual environment
# .venv\Scripts\python.exe -m streamlit run src/app1.py

# Dataset source: https://www.kaggle.com/datasets/notshrirang/spotify-million-song-dataset
streamlit run src/app1.py --server.fileWatcherType none
